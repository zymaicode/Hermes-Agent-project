#### 1. `electron/overlay/aiQuery.ts` — Overlay AI query handler (receives query, collects system snapshot, calls DeepSeek, returns answer)
```typescript
import { hardwareCollector } from '../hardware/collector';

interface OverlayAiResult {
  answer: string;
  error?: string;
}

export async function handleOverlayAiQuery(query: string, config: { endpoint: string; model: string; apiKey: string }): Promise<OverlayAiResult> {
  if (!config.apiKey) {
    return { answer: '❌ 请先在设置中配置 API Key', error: 'no_api_key' };
  }

  try {
    const snapshot = await hardwareCollector.getFullSnapshot();
    const systemInfo = `当前系统状态：\nCPU: ${snapshot.cpu.name} — 使用率 ${snapshot.cpu.usage}% — 温度 ${snapshot.cpu.temp}°C\n内存: ${(snapshot.memory.total / 1024 / 1024 / 1024).toFixed(1)}GB — 占用 ${snapshot.memory.usagePercent}%\nGPU: ${snapshot.gpu.name} — 使用率 ${snapshot.gpu.usage}% — 温度 ${snapshot.gpu.temp}°C\n磁盘: ${snapshot.disks.map((d: any) => `${d.name || d.model}: ${d.usagePercent}%`).join(', ')}`;

    const prompt = `你是一个PC硬件助手。以下是当前系统状态：\n${systemInfo}\n\n用户问：${query}\n\n请基于以上数据简短回答（50字以内），给出具体建议。`;

    const url = `${config.endpoint}/v1/chat/completions`;
    const response = await fetch(url, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${config.apiKey}` },
      body: JSON.stringify({
        model: config.model,
        messages: [{ role: 'user', content: prompt }],
        max_tokens: 256,
      }),
    });

    if (!response.ok) throw new Error(`API error: ${response.status}`);

    const data = await response.json() as { choices: Array<{ message: { content: string } }> };
    return { answer: data.choices[0]?.message?.content?.trim() || '（无回复）' };
  } catch (err: any) {
    return { answer: `❌ 查询失败: ${err.message}`, error: err.message };
  }
}
```

#### 2. Create `electron/overlay/overlay-ai.js` — AI interaction logic for overlay
```javascript
(function() {
  const aiBar = document.getElementById('ai-bar');
  const aiInput = document.getElementById('ai-input');
  const aiResult = document.getElementById('ai-result');
  const aiSend = document.getElementById('ai-send');
  const aiClose = document.getElementById('ai-close');
  let isOpen = false;

  // Listen for toggle from main process
  window.addEventListener('message', (e) => {
    if (e.data.type === 'toggle-ai') {
      isOpen = !isOpen;
      aiBar.style.display = isOpen ? 'flex' : 'none';
      if (!isOpen) {
        aiResult.style.display = 'none';
        aiResult.textContent = '';
      }
      if (isOpen) aiInput.focus();
    }
  });

  // IPC bridge to main process
  const { ipcRenderer } = require('electron');

  async function sendQuery() {
    const query = aiInput.value.trim();
    if (!query) return;
    aiInput.value = '';
    aiResult.style.display = 'block';
    aiResult.innerHTML = '<div class="ai-loading">思考中...</div>';

    try {
      const result = await ipcRenderer.invoke('pchelper:overlay-ai-query', query);
      aiResult.textContent = result.answer || '（无回复）';
    } catch (err) {
      aiResult.textContent = '❌ 查询失败';
    }
  }

  aiSend.addEventListener('click', sendQuery);
  aiInput.addEventListener('keydown', (e) => { if (e.key === 'Enter') sendQuery(); });
  aiClose.addEventListener('click', () => {
    isOpen = false;
    aiBar.style.display = 'none';
    aiResult.style.display = 'none';
    aiResult.textContent = '';
    // Notify main process to restore focus
    ipcRenderer.send('pchelper:overlay-ai-close');
  });
})();
```
