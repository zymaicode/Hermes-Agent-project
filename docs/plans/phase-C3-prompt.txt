## Phase C3: 系统托盘增强

项目：PCHelper (pc-toolkit) — Windows 系统工具，Electron + React + Vite + TypeScript + Zustand + better-sqlite3 + Recharts + lucide-react

### Task

为 PCHelper 添加完善的系统托盘（Tray）功能，包括右键菜单、最小化到托盘、托盘气球提示、以及实时托盘图标（CPU 占用率低/中/高动态变化）。

### Files to create:

#### 1. `electron/trayManager.ts` — 托盘管理器
```typescript
import { Tray, Menu, nativeImage, app, BrowserWindow, Notification } from 'electron';
import path from 'path';

let tray: Tray | null = null;
let mainWindow: BrowserWindow | null = null;
let currentCPU = 0;
let currentMem = 0;

const iconDir = path.join(__dirname, '../assets');

// 根据 CPU 使用率返回不同的图标路径
function getCpuIconPath(usage: number): string {
  const base = path.join(iconDir, 'icon.png');
  return base; // 使用同一个 icon，通过 setImage 刷新
}

// 生成图标 overlay 文本（无法直接叠加，用不同图标代替）
// 使用 icon.png 作为唯一图标，OK

export function initTray(window: BrowserWindow): void {
  mainWindow = window;
  const iconPath = path.join(iconDir, 'icon.png');
  tray = new Tray(iconPath);

  updateTrayMenu();

  tray.setToolTip('PCHelper - 电脑工具箱');

  // 双击托盘图标打开/最小化窗口
  tray.on('double-click', () => {
    if (mainWindow) {
      if (mainWindow.isVisible()) {
        mainWindow.focus();
      } else {
        mainWindow.show();
        mainWindow.focus();
      }
    }
  });
}

function updateTrayMenu(): void {
  if (!tray) return;

  const contextMenu = Menu.buildFromTemplate([
    {
      label: `CPU: ${currentCPU.toFixed(1)}% | Mem: ${currentMem.toFixed(0)}%`,
      enabled: false,
    },
    { type: 'separator' },
    {
      label: '打开主窗口',
      click: () => {
        if (mainWindow) {
          mainWindow.show();
          mainWindow.focus();
        }
      },
    },
    {
      label: '最小化到托盘',
      click: () => {
        if (mainWindow) {
          mainWindow.hide();
        }
      },
    },
    { type: 'separator' },
    {
      label: '退出',
      click: () => {
        app.quit();
      },
    },
  ]);

  tray.setContextMenu(contextMenu);
}

export function updateTrayStats(cpu: number, mem: number): void {
  currentCPU = cpu;
  currentMem = mem;
  updateTrayMenu();
}

export function showTrayNotification(title: string, body: string): void {
  if (!mainWindow) return;
  const notification = new Notification({ title, body, icon: path.join(iconDir, 'icon.png') });
  notification.show();
  notification.on('click', () => {
    if (mainWindow) {
      mainWindow.show();
      mainWindow.focus();
    }
  });
}

export function destroyTray(): void {
  if (tray) {
    tray.destroy();
    tray = null;
  }
}

export function isTrayInitialized(): boolean {
  return tray !== null;
}
```

#### 2. `electron/notifications/notificationManager.ts` — 系统通知管理器
```typescript
import { Notification } from 'electron';
import path from 'path';

// 队列管理，避免短时间内弹太多通知
let lastNotificationTime = 0;
const MIN_INTERVAL_MS = 30000; // 30秒最小间隔

export function sendSystemNotification(title: string, body: string, iconPath?: string): void {
  const now = Date.now();
  if (now - lastNotificationTime < MIN_INTERVAL_MS) return;
  lastNotificationTime = now;

  const notification = new Notification({
    title,
    body,
    icon: iconPath || path.join(__dirname, '../../assets/icon.png'),
    urgency: 'normal',
  });
  notification.show();
}

// 用于硬件告警的静默通知（不触发弹窗，只更新托盘提示）
export function silentNotify(title: string, body: string): void {
  // 由托盘通知统一管理
}
```

```typescript
updateTrayStats(snapshot.cpu.usage, snapshot.memory.usagePercent);

  showTrayNotification('内存使用率过高', `当前使用率：${snapshot.memory.usagePercent.toFixed(0)}%，建议关闭部分程序`);

```
